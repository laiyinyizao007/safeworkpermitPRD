# 04 - 状态管理

## 4.1 创建 Workflow Store

```typescript
// src/stores/workflow.ts
import { defineStore } from 'pinia';
import { NodeExecutor } from '@/engine/NodeExecutor';

export const useWorkflowStore = defineStore('workflow', {
  state: () => ({
    currentNodeId: 'n1',
    executionHistory: [],
    formData: {}
  }),

  actions: {
    async executeNode(nodeId: string, executor: NodeExecutor) {
      this.currentNodeId = nodeId;
      const result = await executor.executeNode(nodeId);

      this.executionHistory.push({
        nodeId,
        timestamp: Date.now(),
        success: result
      });

      return result;
    },

    async nextNode(edges: any[]) {
      const edge = edges.find(e => e.from === this.currentNodeId);
      if (edge) {
        this.currentNodeId = edge.to;
      }
    }
  }
});
```

## 4.2 在组件中使用

```vue

import { useWorkflowStore } from '@/stores/workflow';

const workflowStore = useWorkflowStore();
const executor = inject('nodeExecutor');
```
